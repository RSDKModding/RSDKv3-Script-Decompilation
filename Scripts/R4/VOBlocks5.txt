#alias 70: TYPE_VOBLOCKS5


sub ObjectMain
	Object.Value3=Object.PropertyValue
	Object.Value3<<=1
	Object.Value3+=Oscillation
	Sin(Object.Value2,Object.Value3)
	Object.Value2<<=14
	Object.Value2+=Object.YPos
	Object.Value2&=-65536
	Object.Value2-=Object.Value1
endsub


sub ObjectPlayerInteraction
	TempValue0=Object.YPos
	Object.YPos=Object.Value1
	PlayerObjectCollision(C_BOX,-16,-32,16,32)
	switch CheckResult
	case 1
		Player.YPos+=Object.Value2
		break
	case 4
		if Player.Gravity==0
			CallFunction(PlayerObject_Kill)
		endif
		break
	endswitch
	Object.YPos=TempValue0
endsub


sub ObjectDraw
	Sin(Object.Value1,Object.Value3)
	Object.Value1<<=14
	Object.Value1+=Object.YPos
	Object.Value1&=-65536
	DrawSpriteXY(0,Object.XPos,Object.Value1)
	DrawSpriteXY(1,Object.XPos,Object.Value1)
endsub
