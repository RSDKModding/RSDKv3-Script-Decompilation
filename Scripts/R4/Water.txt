function Water_CreateAirBubble
	if Object.Value1 > 0
		Object.Value1--
		if Object.Value2 > 0
			Object.Value2--
		else
			CreateTempObject(TypeName[AirBubble], 2, Player.XPos, Player.YPos)
			Object[TempObjectPos].DrawOrder = 4
			Object[TempObjectPos].Value3 = -34816
			if Player.Direction == FACING_RIGHT
				Object[TempObjectPos].XPos += 393216
			else
				Object[TempObjectPos].XPos -= 393216
				Object[TempObjectPos].Value1 = 256
			end if
			Object[TempObjectPos].Value2 = Object[TempObjectPos].XPos
			Object.Value2 = 512
		end if
	else
		Rand(TempValue0, 3)
		if TempValue0 == 1
			Rand(Object.Value2, 16)
			Object.Value2 += 8
		else
			Object.Value2 = 512
		end if
		Object.Value1 = 60
		CreateTempObject(TypeName[AirBubble], 2, Player.XPos, Player.YPos)
		Object[TempObjectPos].DrawOrder = 4
		Object[TempObjectPos].Value3 = -34816
		if Player.Direction == FACING_RIGHT
			Object[TempObjectPos].XPos += 393216
		else
			Object[TempObjectPos].XPos -= 393216
			Object[TempObjectPos].Value1 = 256
		end if
		Object[TempObjectPos].Value2 = Object[TempObjectPos].XPos
	end if
end function


function Water_CreateCountBubble
	CreateTempObject(TypeName[CountdownBubble], TempValue0, Player.XPos, Player.YPos)
	Object[TempObjectPos].DrawOrder = 4
	Object[TempObjectPos].Value3 = -34816
	if Player.Direction == FACING_RIGHT

	else
		Object[TempObjectPos].XPos -= 393216
		Object[TempObjectPos].Value1 = 256
	end if
	Object[TempObjectPos].Value2 = Object[TempObjectPos].XPos
end function


sub ObjectMain
	Player.WaterStream = 0
	if Object.YPos < Stage.NewWaterLevel
		Object.YPos += 65536
	end if
	if Object.YPos > Stage.NewWaterLevel
		Object.YPos -= 65536
	end if
	TempValue0 = Oscillation
	TempValue0 <<= 2
	Sin(Stage.WaterLevel, TempValue0)
	Stage.WaterLevel <<= 9
	Stage.WaterLevel += Object.YPos
	if Player.GravityStrength == 14336
		if Player.YPos > Object.YPos
			if Player.State != PlayerObject_Death
				Player.YVelocity >>= 2
				if Object.YPos < Stage.NewWaterLevel
					Player.YVelocity += 65536
				end if
				if Player.State == PlayerObject_HandleTailsFlight
					StopSfx(24)
					StopSfx(25)
				end if
				Player.XVelocity >>= 1
				Player.Speed >>= 1
				Player.TopSpeed >>= 1
				Player.Acceleration >>= 1
				Player.Deceleration >>= 1
				Player.AirAcceleration >>= 1
				Player.AirDeceleration >>= 1
				Player.GravityStrength = 4096
				Player.JumpStrength = 229376
				Player.JumpCap = -131072
				CreateTempObject(TypeName[WaterSplash], 0, Player.XPos, Stage.WaterLevel)
				Object[TempObjectPos].DrawOrder = 4
				PlayStageSfx(0, 0)
				Object.Value1 = 52
				Player.AirTimer = 0
			end if
		end if
	else
		if Player.AirTimer == 0
			Object.State = 0
		end if
		switch Object.State
		case 0
			CallFunction(Water_CreateAirBubble)
			Player.AirTimer++
			if Player.AirTimer == 360
				PlayStageSfx(9, 0)
				Object.State++
			end if
			break
		case 1
			CallFunction(Water_CreateAirBubble)
			Player.AirTimer++
			if Player.AirTimer == 660
				PlayStageSfx(9, 0)
				Object.State++
			end if
			break
		case 2
			CallFunction(Water_CreateAirBubble)
			Player.AirTimer++
			if Player.AirTimer == 960
				PlayStageSfx(9, 0)
				Object.State++
			end if
			break
		case 3
			CallFunction(Water_CreateAirBubble)
			Player.AirTimer++
			if Player.AirTimer == 1080
				TempValue0 = 0
				CallFunction(Water_CreateCountBubble)
				Object.State++
			end if
			break
		case 4
			Player.AirTimer++
			if Player.AirTimer == 1200
				TempValue0 = 1
				CallFunction(Water_CreateCountBubble)
				Object.State++
			end if
			break
		case 5
			CallFunction(Water_CreateAirBubble)
			Player.AirTimer++
			if Player.AirTimer == 1320
				TempValue0 = 2
				CallFunction(Water_CreateCountBubble)
				Object.State++
			end if
			break
		case 6
			CallFunction(Water_CreateAirBubble)
			TempValue0 = Player.AirTimer
			TempValue0 &= 63
			if TempValue0 == 0
				PlayStageSfx(10, 0)
			end if
			Player.AirTimer++
			if Player.AirTimer == 1440
				TempValue0 = 3
				CallFunction(Water_CreateCountBubble)
				Object.State++
			end if
			break
		case 7
			CallFunction(Water_CreateAirBubble)
			TempValue0 = Player.AirTimer
			TempValue0 &= 63
			if TempValue0 == 0
				PlayStageSfx(10, 0)
			end if
			Player.AirTimer++
			if Player.AirTimer == 1560
				TempValue0 = 4
				CallFunction(Water_CreateCountBubble)
				Object.State++
			end if
			break
		case 8
			CallFunction(Water_CreateAirBubble)
			TempValue0 = Player.AirTimer
			TempValue0 &= 63
			if TempValue0 == 0
				PlayStageSfx(10, 0)
			end if
			Player.AirTimer++
			if Player.AirTimer == 1680
				TempValue0 = 5
				CallFunction(Water_CreateCountBubble)
				Object.State++
			end if
			break
		case 9
			CallFunction(Water_CreateAirBubble)
			Player.AirTimer++
			if Player.AirTimer == 1800
				if Player.State != PlayerObject_Death
					Stage.TimeEnabled = 0
					Screen.CameraEnabled = 0
					Player.State = PlayerObject_Drown
					Player.Animation = ANI_DROWNING
					Player.Speed = 0
					Player.XVelocity = 0
					Player.YVelocity = 0
					Player.TileCollisions = 0
					Player.ObjectInteraction = 0
					Player.DrawOrder = 5
					Object.Value1 = 2
					Object.Value3 = 60
					PlayStageSfx(7, 0)
					Object.State++
				end if
			end if
			break
		case 10
			Player.AirTimer++
			if Player.AirTimer == 1860
				Object.State++
			end if
			if Object.Value1 > 0
				Object.Value1--
				if Object.Value2 > 0
					Object.Value2--
				else
					Rand(TempValue1, 2)
					TempValue1 += 2
					CreateTempObject(TypeName[AirBubble], TempValue1, Player.XPos, Player.YPos)
					Object[TempObjectPos].DrawOrder = 5
					Object[TempObjectPos].Value3 = -34816
					Object[TempObjectPos].YPos -= 393216
					Rand(Object[TempObjectPos].Value1, 256)
					Object[TempObjectPos].Value2 = Object[TempObjectPos].XPos
					Object.Value2 = 512
				end if
			else
				Rand(TempValue0, 4)
				if TempValue0 == 1
					Object.Value2 = 2
				else
					Object.Value2 = 512
				end if
				Object.Value1 = 6
				Rand(TempValue1, 2)
				TempValue1 += 2
				CreateTempObject(TypeName[AirBubble], TempValue1, Player.XPos, Player.YPos)
				Object[TempObjectPos].DrawOrder = 5
				Object[TempObjectPos].Value3 = -34816
				Object[TempObjectPos].YPos -= 393216
				Rand(Object[TempObjectPos].Value1, 256)
				Object[TempObjectPos].Value2 = Object[TempObjectPos].XPos
			end if
			break
		end switch
		if Player.YPos < Object.YPos
			if Player.State != PlayerObject_Death
				Player.YVelocity <<= 1
				if Player.YVelocity < -1048576
					Player.YVelocity = -1048576
				end if
				if Player.State == PlayerObject_HandleTailsFlight
					if Player.Timer < 480
						PlaySfx(24, 1)
					else
						PlaySfx(25, 1)
					end if
				end if
				Player.TopSpeed = 393216
				Player.Acceleration = 3072
				Player.Deceleration = 3072
				Player.AirAcceleration = 6144
				Player.AirDeceleration = 1536
				Player.GravityStrength = 14336
				Player.JumpStrength = 425984
				Player.JumpCap = -262144
				CreateTempObject(TypeName[WaterSplash], 0, Player.XPos, Stage.WaterLevel)
				Object[TempObjectPos].DrawOrder = 4
				PlayStageSfx(0, 0)
				Player.AirTimer = 0
			end if
		end if
	end if
	Stage.WaterLevel >>= 16
end sub


sub ObjectDraw
	Object.Value0++
	if Object.Value0 > 23
		Object.Value0 = 0
		Object.Frame++
		Object.Frame &= 3
	end if
	TempValue0 = Screen.XOffset
	Sin(TempValue1, Oscillation)
	TempValue1 >>= 5
	TempValue0 += TempValue1
	TempValue0 &= 63
	FlipSign(TempValue0)
	TempValue1 = Stage.WaterLevel
	TempValue1 -= Screen.YOffset
	DrawSpriteScreenFX(Object.Frame, FX_INK, TempValue0, TempValue1)
	TempValue0 += 128
	DrawSpriteScreenFX(Object.Frame, FX_INK, TempValue0, TempValue1)
	TempValue0 += 128
	DrawSpriteScreenFX(Object.Frame, FX_INK, TempValue0, TempValue1)
	TempValue0 += 128
	DrawSpriteScreenFX(Object.Frame, FX_INK, TempValue0, TempValue1)
end sub
