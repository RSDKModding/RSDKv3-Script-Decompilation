#alias 45: TYPE_MECHABU


sub ObjectMain
	switch Object.State
	case 0
		ObjectTileCollision(0,0,15,0)
		if CheckResult==1
			Object.State=1
		else
			Object.YPos+=65536
		endif
		break
	case 1
		if Object.Value0<416
			Object.Value0++
			Object.XPos+=20480
		else
			Object.Value0=0
			Object.State=2
		endif
		ObjectTileGrip(0,0,15,0)
		break
	case 2
		if Object.Value0<416
			Object.Value0++
			Object.XPos-=20480
		else
			Object.Value0=0
			Object.State=1
		endif
		ObjectTileGrip(0,0,15,0)
		break
	endswitch
	CallFunction(StageSetup_CheckGoodFuture)
endsub


sub ObjectPlayerInteraction
	if Object.State<3
		PlayerObjectCollision(C_TOUCH,-16,-14,16,14)
		if CheckResult==1
			CallFunction(PlayerObject_BadnikBreak)
		endif
		if Object.Type==TypeName[MechaBu]
			if Object.Value4>59
				PlayerObjectCollision(C_TOUCH,-30,-32,-8,0)
				if CheckResult==1
					CallFunction(PlayerObject_Hit)
				endif
			else
				PlayerObjectCollision(C_TOUCH,-36,-16,-14,16)
				if CheckResult==1
					CallFunction(PlayerObject_Hit)
				endif
			endif
		endif
	endif
endsub


sub ObjectDraw
	Object.Value5++
	if Object.Value5==48
		Object.Value5=0
		PlayStageSfx(1,0)
	endif
	TempValue0=Object.AnimationTimer
	TempValue0/=3
	Object.AnimationTimer++
	Object.AnimationTimer%=6
	if Object.PropertyValue==0
		if Object.Value4>59
			TempValue0+=5
			DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
			DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
			DrawSpriteFX(4,FX_FLIP,Object.XPos,Object.YPos)
		else
			TempValue0+=2
			DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
			DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
			DrawSpriteFX(1,FX_FLIP,Object.XPos,Object.YPos)
		endif
		Object.Value4++
		Object.Value4%=120
	else
		TempValue0+=5
		DrawSpriteFX(TempValue0,FX_FLIP,Object.XPos,Object.YPos)
		DrawSpriteFX(0,FX_FLIP,Object.XPos,Object.YPos)
		DrawSpriteFX(4,FX_FLIP,Object.XPos,Object.YPos)
	endif
endsub
