#alias 37: TYPE_STATICROCK


sub ObjectPlayerInteraction
	switch Object.PropertyValue
	case 0
		PlayerObjectCollision(C_BOX,-16,-16,16,16)
		break
	case 1
		PlayerObjectCollision(C_TOUCH,-16,-14,16,16)
		if CheckResult==1
			if Player.XPos<Object.XPos
				Player.XPos=Player.CollisionLeft
				Player.XPos<<=16
				Player.XPos+=Object.XPos
				Player.XPos-=1048576
			else
				Player.XPos=Player.CollisionRight
				Player.XPos<<=16
				Player.XPos+=Object.XPos
				Player.XPos+=1048576
			endif
		endif
		break
	case 2
		if Player.CollisionMode==1
			PlayerObjectCollision(C_TOUCH,-16,-16,16,16)
			if CheckResult==1
				Player.YPos=Player.CollisionBottom
				Player.YPos<<=16
				Player.YPos+=Object.YPos
				Player.YPos+=1048576
				Player.Speed=0
			endif
		endif
		if Player.CollisionMode==3
			PlayerObjectCollision(C_TOUCH,-16,-16,16,16)
			if CheckResult==1
				Player.YPos=Player.CollisionBottom
				Player.YPos<<=16
				Player.YPos+=Object.YPos
				Player.YPos+=1048576
				Player.Speed=0
			endif
		endif
		break

	endswitch
endsub


sub ObjectDraw
	DrawSprite(Object.PropertyValue)
endsub
