	switch Object.State
	case 0
		Object.Value0++
		Object.Value0%=40
		break
	case 1
		if Object.Value1<41
			Object.Value1++
		else
			Object.Value1=0
			Object.Value0=0
			if Object[+1].State<5
				Object.State=0
			else
				Object.State=2
			endif
		endif
		break
	case 2
		Object.Value0++
		Object.Value0&=31
		break
	case 3
		Object.Value0++

		break
	endswitch
endsub
