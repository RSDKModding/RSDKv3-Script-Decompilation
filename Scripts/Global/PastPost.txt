sub ObjectPlayerInteraction
	if Object.State==0
		if Player.CollisionPlane==Object.Value1
			PlayerObjectCollision(C_TOUCH,-8,-24,8,32)
			if CheckResult==1
				Warp.Destination=1
				Object.State=1
				Object.Priority=1
				Object.Frame=1
				PlaySfx(13,0)
#platform: Use_Haptics
				HapticEffect(20,0,0,0)
#endplatform
			endif
		endif
	endif
endsub


sub ObjectDraw
	switch Object.State
	case 0
		DrawSprite(0)
		DrawSprite(1)
		break
	case 1
		DrawSprite(0)
		Object.Value0++
		if Object.Value0==64
			Object.Value0=0
			Object.State=2
			Object.Priority=0
		endif
		DrawSprite(Object.Frame)
		TempValue0=Object.Value0
		TempValue0&=1
		Object.Frame+=TempValue0
		if Object.Frame>8
			Object.Frame=1
		endif
		break
	case 2
		DrawSprite(0)
		DrawSprite(5)
		break
	endswitch
endsub
