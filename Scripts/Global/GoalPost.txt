// States
#alias 0	:	GOALPOST_IDLE
#alias 1	:	GOALPOST_BLANK

// Warp Destination
#alias 0	:	NONE

sub ObjectMain
	if Object.State == GOALPOST_IDLE
		TempValue0 = Object.YPos
		TempValue0 += 6291456
		if Player.YPos < TempValue0
			if Player.XPos > Object.XPos
				Stage.NewXBoundary1 = Object.XPos
				Stage.NewXBoundary1 >>= 16
				Stage.NewXBoundary1 -= 160
				Warp.Destination = NONE
				Warp.Timer = 0
				Object.State = GOALPOST_BLANK
				if Object.PropertyValue == 1
					PlayerObjectCollision(C_TOUCH, -32, -256, 32, 16)
					if CheckResult == true
						// Take The High Road Unlock Criteria
						if Stage.PlayerListPos == 0
							if Stage.DebugMode == 0
								SetAchievement(3, 100)
							end if
						end if
					end if
				end if
			end if
		end if
	end if
end sub


sub ObjectDraw
	DrawSprite(0)
end sub


sub ObjectStartup
	if Stage.PlayerListPos == 0		// PlayerName[SONIC]
		LoadSpriteSheet("Global/Items2.gif")
	else
		LoadSpriteSheet("Global/Items2_t.gif")
	end if
	SpriteFrame(-20, -44, 40, 56, 34, 165)	// #0 - Goal Post
end sub


sub RSDKEdit
	if Editor.ReturnVariable == true
		switch Editor.VariableID
		case EDIT_VAR_PROPVAL // property value
			CheckResult = Object.PropertyValue
			break
		case 0 // achievmentEnabled
			CheckResult = Object.PropertyValue
			break
		end switch
	else
		switch Editor.VariableID
		case EDIT_VAR_PROPVAL // property value
			Object.PropertyValue = Editor.VariableValue
			break
		case 0 // achievmentEnabled
			Object.PropertyValue = Editor.VariableValue
			break
		end switch
	end if
end sub

sub RSDKDraw
	DrawSprite(0)
end sub


sub RSDKLoad
	LoadSpriteSheet("Global/Items2.gif")
	SpriteFrame(-20, -44, 40, 56, 34, 165)

	AddEditorVariable("achievmentEnabled")
	SetActiveVariable("achievmentEnabled")
	AddEnumVariable("True", 0)
	AddEnumVariable("False", 1)
end sub
