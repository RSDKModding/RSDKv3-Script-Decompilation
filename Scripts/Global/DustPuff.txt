sub ObjectDraw
	if Object.Frame > 3
		// Move to the correct player slot and follow that player
		// Do note, though, that CD only ever has 1 player active so this is pretty much the same as simply moving to P1's slot
		Stage.ActivePlayer = Object.PropertyValue

		Object.Direction = Player.Direction
		Object.XPos = Player.XPos
		Object.iYPos = Player.CollisionBottom
		Object.YPos += Player.YPos

		Stage.ActivePlayer = 0

		if Player.Animation != ANI_SPINDASH
			Object.Type = TypeName[BlankObject]
		else
			DrawSpriteFX(Object.Frame, FX_FLIP, Object.XPos, Object.YPos)
		end if
	else
		DrawSpriteFX(Object.Frame, FX_FLIP, Object.XPos, Object.YPos)
	end if

	switch Object.Frame
	case 0
	case 1
	case 2
		Object.AnimationTimer++
		if Object.AnimationTimer > 3
			Object.Frame++
			Object.AnimationTimer = 0
		end if
		break

	case 3
		Object.AnimationTimer++
		if Object.AnimationTimer > 3
			Object.Type = TypeName[BlankObject]
		end if
		break

	case 4
	case 5
	case 6
	case 7
	case 8
	case 9
		Object.AnimationTimer++
		if Object.AnimationTimer > 1
			Object.Frame++
			Object.AnimationTimer = 0
		end if
		break

	case 10
		Object.AnimationTimer++
		if Object.AnimationTimer > 1
			Object.Frame = 4
			Object.AnimationTimer = 0
		end if
		break
		
	end switch
end sub
