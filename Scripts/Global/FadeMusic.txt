// Aliases
#alias 60: TYPE_FADEMUSIC
#alias Object.Value0	:	Object.Timer

// Property Values
#alias 0				:	FADEMUSIC_FADE_TO_BOSS
#alias 1				:	FADEMUSIC_FADE_TO_LEVEL
#alias 2				:	FADEMUSIC_SSZEGGMAN
#alias 3				:	FADEMUSIC_SSZAMY

// Priority
#alias 1				:	PRIORITY_ACTIVE

sub ObjectMain
	switch Object.PropertyValue
	case FADEMUSIC_FADE_TO_BOSS
		if Object.Timer < 100
			Object.Timer++
			Music.Volume--
		else
			PlayMusic(4)
			Object.Type = TypeName[BlankObject]
		end if
		break

	case FADEMUSIC_FADE_TO_LEVEL
		if Object.Timer < 100
			Object.Timer++
			Music.Volume--
		else
			PlayMusic(0)
			Object.Type = TypeName[BlankObject]
		end if
		break

	case FADEMUSIC_SSZEGGMAN
		if Object.Timer < 50
			Object.Timer++
			Music.Volume -= 2
		else
			PlayMusic(0)
			Object.Type = TypeName[BlankObject]
		end if
		break

	case FADEMUSIC_SSZAMY
		if Object.Timer < 100
			Object.Timer++
			Music.Volume--
		else
			PlayMusic(1)
			Object.Type = TypeName[BlankObject]
			Object[30].Type = TypeName[ActFinish]
			Object[30].DrawOrder = 6
			Object[30].Priority = PRIORITY_ACTIVE
		end if
		break

	end switch
end sub


sub RSDKDraw
	DrawSprite(0)
end sub


sub RSDKLoad
	LoadSpriteSheet("Global/Display.gif")
	SpriteFrame(-16, -16, 32, 32, 1, 143)

	// Although used by the object, it shouldn't be set by the editor
	SetVariableAlias(ALIAS_VAR_PROPVAL, "unused")
end sub
