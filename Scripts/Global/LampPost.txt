sub ObjectPlayerInteraction
	if Object.State==0
		PlayerObjectCollision(C_TOUCH,-8,-32,8,32)
		if CheckResult==1
			LampPost.Check=Object.EntityNo
			if Player.XPos<Object.XPos
				LampPost.Side=-1572864
#platform: Mobile			  
				ObjectTileCollision(2,-24,0,0)
				if CheckResult==1
					LampPost.Side=0
#endplatform
			else
				LampPost.Side=1572864
#platform: Mobile
				ObjectTileCollision(1,24,0,0)
				if CheckResult==1
					LampPost.Side=0
				endif
#endplatform
			endif
			Rec_Milliseconds=Stage.MilliSeconds
			Rec_Seconds=Stage.Seconds
			Rec_Minutes=Stage.Minutes
			Object.State=1
			Object.Value0=384
			PlaySfx(9,0)
#platform: Use_Haptics
			HapticEffect(20,0,0,0)
#endplatform		 
		endif
	endif
endsub


sub ObjectDraw
	switch Object.State
	case 0
		DrawSprite(0)
		break
	case 1
		DrawSprite(1)
		Cos(TempValue0,Object.Value0)
		TempValue0<<=10
		Sin(TempValue1,Object.Value0)
		TempValue1<<=10
		TempValue0+=Object.XPos
		TempValue1+=Object.YPos
		TempValue1-=1572864
		DrawSpriteXY(2,TempValue0,TempValue1)
		Object.Value0+=16
		if Object.Value0>896
			Object.Value0=0
			Object.State=2
		endif
		break
	case 2
		DrawSprite(1)
		Object.Value0++
		if Object.Value0==8
			Object.Value0=0
		endif
		TempValue0=Object.YPos
		TempValue0-=2097152
		if Object.Value0>3
			DrawSpriteXY(4,Object.XPos,TempValue0)
		else
			DrawSpriteXY(3,Object.XPos,TempValue0)
		endif
		break

	endswitch
endsub
