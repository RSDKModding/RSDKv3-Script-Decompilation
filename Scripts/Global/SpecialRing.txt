// Aliases
#alias Player.Value0	:	Player.Rings

// States
#alias 0				:	SPECIALRING_IDLE
#alias 1				:	SPECIALRING_BLANK
// Game Modes
#alias 2				:	GAMEMODE_TIMEATTACK

sub ObjectMain
	if Object.State == SPECIALRING_IDLE
		if Player.Rings > 49
			PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
			if CheckResult == true
				Object[0].Type = TypeName[BlankObject]
				Object[1].Type = TypeName[BlankObject]
				Player.ControlMode = -1
				Player.Direction = FACING_RIGHT
				ResetObjectEntity(2, TypeName[BlankObject], 0, 0, 0)
				Object.DrawOrder = 4
				Object.State = SPECIALRING_BLANK
#platform: Standard
				Object.Frame = 16
#endplatform
#platform: Mobile
				Object.Frame = 32
				StopSfx(24)
				StopSfx(25)
#endplatform
				PlaySfx(18, 0)
				Player.Left = 0
				Player.Right = 0
#platform: Use_Haptics
				HapticEffect(40, 0, 0, 0)
#endplatform
			end if
		end if
	end if
end sub


sub ObjectDraw
	switch Object.State
	case SPECIALRING_IDLE
		if Player.Rings > 49
			TempValue0 = Object.Frame
#platform: Standard
			TempValue0 >>= 3
			DrawSprite(TempValue0)
#endplatform

#platform: Mobile
			TempValue0 >>= 2
			if TempValue0 > 3
				Object.Direction = FACING_LEFT
			else
				Object.Direction = FACING_RIGHT
			end if
			DrawSpriteFX(TempValue0, FX_FLIP, Object.XPos, Object.YPos)
#endplatform

			Object.Frame++
			if Object.Frame > 31
				Object.Frame = 0
			end if
		end if
		break
	case SPECIALRING_BLANK
		TempValue0 = Object.Frame
		TempValue0 >>= 2

#platform: Mobile
		if TempValue0 > 11
			Object.Direction = FACING_LEFT
		else
			Object.Direction = FACING_RIGHT
		end if
#endplatform

		DrawSpriteFX(TempValue0, FX_FLIP, Object.XPos, Object.YPos)

#platform: Standard
		Object.Frame++
		CheckGreater(Object.Frame,31)
		if CheckResult == true
			Object.Direction = FACING_LEFT
		end if
#endplatform

#platform: Mobile
		CheckEqual(TempValue0,11)
		if CheckResult == true
			Object.Direction = FACING_LEFT
				DrawSpriteFX(TempValue0, FX_FLIP, Object.XPos, Object.YPos)
		else
			if TempValue0 == 15
				Object.Direction = FACING_LEFT
				DrawSpriteFX(TempValue0, FX_FLIP, Object.XPos, Object.YPos)
			end if
		end if
#endplatform

#platform: Standard
		CheckGreater(Object.Frame,47)
#endplatform

#platform: Mobile
		Object.Frame++
		CheckGreater(Object.Frame,63)
#endplatform

		if CheckResult == true
			ResetObjectEntity(Object.EntityNo, TypeName[BlankObject], 0, 0, 0)
		end if
		break
	end switch
end sub
