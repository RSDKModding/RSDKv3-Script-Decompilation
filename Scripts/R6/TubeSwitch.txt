#alias 58: TYPE_TUBESW
#alias Player.Value6:Player.MinRollSpeed


sub ObjectPlayerInteraction
	PlayerObjectCollision(C_TOUCH, Object.Value2, Object.Value2, Object.Value3, Object.Value3)
	if CheckResult == 1
		switch Object.PropertyValue
		case 0
			if Player.Speed > -1
				Player.Direction = FACING_RIGHT
				Player.Speed = 655360
				Player.State = PlayerObject_Tunnel
				Player.Animation = ANI_JUMPING
				Player.MinRollSpeed = 655360
			else
				Player.State = PlayerObject_HandleRolling
				Player.Animation = ANI_JUMPING
			end if
			break
		case 1
			if Player.CollisionMode == 0
				if Player.Speed < 1
					Player.Direction = FACING_LEFT
					Player.Speed = -655360
					Player.State = PlayerObject_Tunnel
					Player.Animation = ANI_JUMPING
					Player.MinRollSpeed = 655360
				else
					Player.State = PlayerObject_HandleRolling
					Player.Animation = ANI_JUMPING
				end if
			else
				Player.Gravity = 1
				FlipSign(Player.Speed)
				Player.XVelocity = Player.Speed
				Player.State = PlayerObject_HandleAir
			end if
			break
		case 2
			Player.Gravity = 1
			Player.CollisionMode = 0
			Player.XVelocity = 0
			Player.Speed = 0
			break
		case 3
			if Player.YVelocity < 1
				Player.State = PlayerObject_Tunnel
				Player.MinRollSpeed = 786432
				Player.Speed = -786432
				Player.Gravity = 0
				Player.Angle = 64
				Player.CollisionMode = 3
				Player.Animation = ANI_JUMPING
				Screen.AdjustCameraY = 0
			else
				if Player.State != PlayerObject_HandleTailsFlight
					Player.Animation = ANI_JUMPING
				end if
			end if
			break
		case 4
			if Player.State == PlayerObject_Tunnel
				Player.State = PlayerObject_HandleRolling
			end if
			break
		case 5
			if Player.State == PlayerObject_StartTTCtrlLock
				if Player.JumpPress == 1
					Object.Value1 = 1
				else
					if Player.JumpHold == 0
						Object.Value1 = 0
					end if
				end if
				if Object.Value1 == 1
					PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
					if CheckResult == 1
						Player.State = PlayerObject_HandleAir
						Player.Gravity = 1
						Player.XPos = Object.XPos
						Player.YPos = Object.YPos
						Player.Speed = -524288
						Player.XVelocity = -524288
						Player.YVelocity = -262144
						Object.Value1 = 0
						Object.Value4 = 1
					end if
				end if
			else
				PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
				if CheckResult == 1
					if Object.Value4 == 0
						Player.Gravity = 0
						Player.XPos = Object.XPos
						Player.YPos = Object.YPos
						Player.XVelocity = 0
						Player.YVelocity = 0
						Player.Speed = 524288
						Player.State = PlayerObject_StartTTCtrlLock
						Player.Animation = ANI_JUMPING
						Player.MinRollSpeed = 524288
						Player.Angle = 224
						Player.CollisionMode = 0
						PlaySfx(7, 0)
						Object.Value4 = 1
					end if
				else
					Object.Value4 = 0
				end if
			end if
			break
		case 6
			if Player.State == PlayerObject_StartTTCtrlLock
				if Player.JumpPress == 1
					Object.Value1 = 1
				else
					if Player.JumpHold == 0
						Object.Value1 = 0
					end if
				end if
				if Object.Value1 == 1
					PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
					if CheckResult == 1
						Player.State = PlayerObject_HandleAir
						Player.Gravity = 1
						Player.XPos = Object.XPos
						Player.YPos = Object.YPos
						Object.Value1 = 0
						Player.Speed = 524288
						Player.XVelocity = 524288
						Player.YVelocity = -262144
						Object.Value4 = 1
					end if
				end if
			else
				PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
				if CheckResult == 1
					if Object.Value4 == 0
						Player.Gravity = 0
						Player.XPos = Object.XPos
						Player.YPos = Object.YPos
						Player.XVelocity = 0
						Player.YVelocity = 0
						Player.Speed = 524288
						Player.State = PlayerObject_StartTTCtrlLock
						Player.Animation = ANI_JUMPING
						Player.MinRollSpeed = 524288
						Player.Angle = 36
						Player.CollisionMode = 3
						PlaySfx(7, 0)
						Object.Value4 = 1
					end if
				else
					Object.Value4 = 0
				end if
			end if
			break
		case 7
			if Object.Value0 == 0
				if Player.State == PlayerObject_StartTTCtrlLock
					FlipSign(Player.Speed)
					if Player.CollisionMode == 0
						Player.CollisionMode = 2
						Player.Angle = 128
					else
						Player.CollisionMode = 0
						Player.Angle = 0
					end if
				end if
			end if
			break
		case 8
			if Player.State == PlayerObject_StartTTCtrlLock
				if Player.JumpPress == 1
					Object.Value1 = 1
				else
					if Player.JumpHold == 0
						Object.Value1 = 0
					end if
				end if
				if Object.Value1 == 1
					PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
					if CheckResult == 1
						Player.State = PlayerObject_HandleAir
						Player.Gravity = 1
						Player.XPos = Object.XPos
						Player.YPos = Object.YPos
						Player.Speed = -524288
						Player.XVelocity = -524288
						Player.YVelocity = 131072
						Object.Value1 = 0
						Object.Value4 = 1
					end if
				end if
			else
				PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
				if CheckResult == 1
					if Object.Value4 == 0
						Player.Speed = -524288
						Player.Gravity = 0
						Player.XPos = Object.XPos
						Player.YPos = Object.YPos
						Player.XVelocity = 0
						Player.YVelocity = 0
						Player.State = PlayerObject_StartTTCtrlLock
						Player.Animation = ANI_JUMPING
						Player.MinRollSpeed = 524288
						Player.Angle = 172
						Player.CollisionMode = 1
						PlaySfx(7, 0)
						Object.Value4 = 1
					end if
				else
					Object.Value4 = 0
				end if
			end if
			break
		case 9
			if Player.YVelocity < 1
				Player.State = PlayerObject_Tunnel
				Player.MinRollSpeed = 786432
				Player.Speed = 786432
				Player.Gravity = 0
				Player.Angle = 192
				Player.CollisionMode = 1
				Player.Animation = ANI_JUMPING
				Screen.AdjustCameraY = 0
			else
				Player.Animation = ANI_JUMPING
			end if
			break
		case 10
			if Object.Value0 == 0
				if Player.State == PlayerObject_HandleRolling
					Player.State = PlayerObject_HandleAir
					Player.Gravity = 1
					Player.Speed = 0
					Player.XVelocity = 0
				end if
				if Player.State == PlayerObject_Tunnel
					Player.State = PlayerObject_HandleRolling
				end if
			end if
			break
		case 11
			if Player.State != PlayerObject_StartTTCtrlLock
				if Object.Value0 == 0
					Player.Gravity = 0
					Player.XPos = Object.XPos
					Player.YPos = Object.YPos
					Player.XVelocity = 0
					Player.YVelocity = 0
					Player.Speed = 524288
					Player.State = PlayerObject_StartTTCtrlLock
					Player.Animation = ANI_JUMPING
					Player.MinRollSpeed = 524288
					Player.Angle = 224
					Player.CollisionMode = 0
					PlaySfx(7, 0)
				end if
			end if
			break
		case 12
			if Player.YVelocity > -1
				Player.State = PlayerObject_Tunnel
				Player.MinRollSpeed = 786432
				Player.Speed = -786432
				Player.Gravity = 0
				Player.Angle = 192
				Player.CollisionMode = 1
				Player.Animation = ANI_JUMPING
				Screen.AdjustCameraY = 0
			else
				Player.State = PlayerObject_Tunnel
				Player.MinRollSpeed = 786432
				Player.Speed = Player.YVelocity
				FlipSign(Player.Speed)
				Player.Gravity = 0
				Player.Angle = 192
				Player.CollisionMode = 1
				Player.Animation = ANI_JUMPING
			end if
			break
		end switch
		Object.Value0 = 1
	else
		Object.Value0 = 0
	end if
end sub


sub ObjectStartup
	ArrayPos0 = 32
	while ArrayPos0 < 1056
		if Object[ArrayPos0].Type == TypeName[TubeSW]
			switch Object[ArrayPos0].PropertyValue
			case 0
			case 1
			case 2
			case 3
			case 4
			case 7
				Object[ArrayPos0].Value2 = -16
				Object[ArrayPos0].Value3 = 16
				break
			case 5
			case 6
			case 8
				Object[ArrayPos0].Value2 = -48
				Object[ArrayPos0].Value3 = 48
				break
			case 9
			case 10
				Object[ArrayPos0].Value2 = -8
				Object[ArrayPos0].Value3 = 8
				break
			case 11
				Object[ArrayPos0].Value2 = -16
				Object[ArrayPos0].Value3 = 16
				break
			end switch
		end if
		ArrayPos0++
	loop
end sub
