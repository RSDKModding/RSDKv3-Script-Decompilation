function MovingBlocks_Function50
	switch Object.Value6
	case 0
		Object.Value1-=32768
		if Object.AnimationTimer==64
			Object.Value3=Object.Value1
		endif
		break
	case 1
		Object.Value1-=32768
		break
	case 2
		Object.Value1+=32768
		Object.Value5+=32768
		if Object.AnimationTimer==64
			Object.Value1=Object.Value5
		endif
		break
	case 3
		Object.Value1+=32768
		Object.Value3+=32768
		break
	case 4
		Object.Value1-=32768
		Object.Value3-=32768
		if Object.AnimationTimer==64
			Object.Value1=Object.Value3
		endif
		break
	case 5
		Object.Value1-=32768
		Object.Value5-=32768
		break

	endswitch
endfunction


sub ObjectMain
	Object.Rotation-=2
	if Object.Rotation<0
		Object.Rotation+=512
	endif
	Object.AnimationTimer++
	CallFunction(MovingBlocks_Function50)
	if Object.AnimationTimer==64
		Object.AnimationTimer=0
		Object.Value6++
		if Object.Value6>5
			Object.Value6=2
		endif
	endif
endsub
