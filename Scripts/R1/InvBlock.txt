sub ObjectPlayerInteraction
	if Object.State<2
		PlayerObjectCollision(C_PLATFORM,-12,-12,12,12)
		Object.Value0=CheckResult
	endif
	if Object[24].PropertyValue>1
		switch Object.State
		case 0
			if Object.Value0==1
				Object.State=1
				Object.Priority=1
				Player.YPos+=131072
			endif
			break
		case 1
			if Object.Value0==1
				Player.YPos+=131072
			endif
			break
		endswitch
	endif
endsub


sub ObjectDraw
	switch Object.State
	case 1
		Object.YPos+=131072
		if Object.OutOfBounds==1
			Object.State=2
			Object.YPos=Object.Value2
		endif
		break
	case 2
		if Object.OutOfBounds==1
			Object.State=0
			Object.Priority=0
		endif
		break
	endswitch
	switch Object[24].PropertyValue
	case 0
		if Object.Value0==1
			if Object.Frame<24
				Object.Frame++
			endif
		else
			if Object.Frame>0
				Object.Frame--
			endif
		endif
		DrawSprite(Object.Frame)
		break
	case 1
		DrawSprite(5)
		break
	case 2
	case 3
		if Object.State<2
			DrawSprite(5)
		endif
		break
	endswitch
endsub
