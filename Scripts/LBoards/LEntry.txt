sub ObjectMain

	switch Object.State
	case 1
		Object.Value5=Object.XPos
		Object.Priority=1
		Object.State++
		break
	case 2
		Object.XPos-=0x300000
		if Object.Value0<25
			Object.Value0++
			if Object.PropertyValue<4
				if Object.Value0==5
					Object[+1].State=1
					if Object.PropertyValue==2
						Leaderboard.ID++
						Leaderboard.ID%=22
						LoadOnlineMenu(1)
					endif
				endif
			endif
		else
			Object.Value0=0
			Object.iXPos=Screen.XSize
			Object.XPos+=0x100000
			Object.State++
		endif
		break
	case 3
		Object.XPos-=0x300000
		if Object.XPos<Object.Value5
			Object.XPos=Object.Value5
			Object.State=0
			if Object.PropertyValue==4
				Object[0].State=2
			endif
		endif
		break
	case 4
		Object.Value5=Object.XPos
		Object.Priority=1
		Object.State++
		break
	case 5
		Object.XPos+=0x300000
		if Object.Value0<25
			Object.Value0++
			if Object.PropertyValue<4
				if Object.Value0==5
					Object[+1].State=4
					if Object.PropertyValue==2
						Leaderboard.ID--
						if Leaderboard.ID<0
							Leaderboard.ID=21
						endif
						LoadOnlineMenu(1)
					endif
				endif
			endif
		else
			Object.Value0=0
			Object.iXPos=-232
			Object.XPos-=0x100000
			Object.State++
		endif
		break
	case 6
		Object.XPos+=0x300000
		if Object.XPos>Object.Value5
			Object.XPos=Object.Value5
			Object.State=0
			if Object.PropertyValue==4
				Object[0].State=2
			endif
		endif
		break
	case 7
		if Object.Value4<95
			Object.Value4+=5
		endif
		Object.Alpha=255
		Object.State++
		break
	case 8
		Object.InkEffect=2
		Object.Alpha=224
		Object.State++
		break
	case 9
		if Object.Alpha>0
			Object.Alpha-=32
			if Object.PropertyValue<4
				if Object.Alpha==128
					Object[+1].State=7
				endif
			endif
		else
			Object.State=0
			if Object.PropertyValue==4
				Object[0].State=2
			endif
		endif
		break
	case 10
		if Object.Value4>0
			Object.Value4-=5
		endif

		Object.State++
		break
	case 11
