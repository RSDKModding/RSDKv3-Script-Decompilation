sub ObjectMain
	if Object.State==1
		Object.Value1+=14336
		Object.YPos+=Object.Value1
		if Object.Value1>=0
			ObjectTileCollision(0,0,16,0)
			if CheckResult==1
				Object.Value1=0
				Object.State=0
			endif
		endif
	endif
endsub


sub ObjectPlayerInteraction
	if Player.CollisionPlane==Object.Value2
		if Player.YVelocity>=0
			if Player.Animation==ANI_JUMPING
				PlayerObjectCollision(C_TOUCH,-16,-14,16,16)
				if CheckResult==1
					Object.State=0
					CreateTempObject(TypeName[SmokePuff],0,Object.XPos,Object.YPos)
					Object[TempObjectPos].DrawOrder=4
					FlipSign(Player.YVelocity)
					if Player.YVelocity>-131072
						Player.YVelocity=-131072
					endif
					Object.Type=TypeName[BrokenMonitor]
					Object.Priority=1
					Object.Alpha=255
					Object.Value0=Object.YPos
					Object.Value1=-196608
					PlaySfx(8,0)
				endif
			else
				PlayerObjectCollision(C_BOX,-15,-14,15,16)
			endif
		else
			PlayerObjectCollision(C_BOX,-15,-16,15,16)
			if CheckResult==4
				Object.State=1
				Object.Value1=-131072
				Player.YVelocity=131072
			endif
		endif
	endif
endsub


sub ObjectDraw
	Object.Frame++
	if Object.Frame>17
		Object.Frame=0
	endif
	switch Object.Frame
	case 0
	case 1
		DrawSprite(0)
		break
	case 6
	case 7
		DrawSprite(9)
		break
	case 12
	case 13
		DrawSprite(10)
		break
	default
		DrawSprite(Object.PropertyValue)
		endswitch
endsub
