sub ObjectMain
	if Object.State==1
		if Object.Value0<39
			Object.Value0++
		else
			Object.State=0
			Object.Value0=0
			Object.Priority=0
		endif
		PathSwap_Frame=Object.Value0
		PathSwap_Frame/=5
	endif
	if PathSwap_Flag==0
		TempValue0=Object.Value5
	else
		TempValue0=Object.Value6
	endif
	if Object.PropertyValue!=1
		SetTileLayerEntry(TempValue0,0,Object.Value1,Object.Value2)
		TempValue0++
		SetTileLayerEntry(TempValue0,0,Object.Value3,Object.Value2)
		TempValue0++
		SetTileLayerEntry(TempValue0,0,Object.Value1,Object.Value4)
		TempValue0++
		SetTileLayerEntry(TempValue0,0,Object.Value3,Object.Value4)
	endif
endsub


sub ObjectPlayerInteraction
	if Object.PropertyValue<2
		if Object.State==0
			if PathSwap_Frame==0
				PlayerObjectCollision(C_TOUCH,-16,-16,16,16)
				if CheckResult==1
					if Object.Value7==0
						Object.State=1
						Object.Priority=1
						Object.Value7=1
						PathSwap_Flag++
						PathSwap_Flag&=1
					endif
				else
					Object.Value7=0
				endif
			endif
		endif
	endif
endsub
