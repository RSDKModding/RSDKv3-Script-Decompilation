sub ObjectMain

		endif

			endif
			PlaySfx(23,0)
		endif

			if Object.Value1>3
				Object.Value1=0
			endif
			PlaySfx(23,0)
		endif
		CheckResult=0

			CheckResult=1
		endif
		if KeyPress.ButtonA==1
			CheckResult=1
		endif
		if CheckResult==1

				ArrayPos0=Object.Value1
				Object[+ArrayPos0].Value2=1
				Object.State++
				switch Object.Value1
				case 0
					StopMusic()
					break
				case 1
					break
				case 2
					StopMusic()
					break
				case 3
					break

				endswitch
				PlaySfx(27,0)
			endif
		endif
		if KeyPress.ButtonB==1
			Object.Value0=248
			Object.State=11
			PlayStageSfx(1,0)
		endif
		break
	case 2
		if Object.Value0<30
			Object.Value0++
		else
			Object.Value0=0
			ArrayPos0=Object.Value1
			Object[+ArrayPos0].Value2=0
			switch Object.Value1
			case 0
				Object.State=3
				break
			case 1
				ReadSaveRAM()
				if SaveRAM[32]==0
					SaveRAM[33]=100
					SaveRAM[34]=100
					SaveRAM[35]=0
				endif
				Object.State=4
				Object.YPos=40
				Object.YPos*=Object.Value1
				Object.YPos+=Screen.CenterY
				Object.YPos-=87
				break
			case 2
				Object.State=3
				break
			case 3
				Object.State=12
				Object.YPos=40
				Object.YPos*=Object.Value1
				Object.YPos+=Screen.CenterY
				Object.YPos-=87
				break

			endswitch
		endif
		break
	case 3
		if Object.Value0<320
			Object.Value0+=8
		else
			switch Object.Value1
			case 0
				Stage.ListPos=6
				LoadStage()
				break
			case 2
				Stage.ListPos=7
				LoadStage()
				break
			case 3
				Engine.State=RESET_GAME
				break

			endswitch
		endif
		TempValue0=Object.Value0
		if TempValue0>255
			TempValue0=255
		endif
		SetScreenFade(0,0,0,TempValue0)
		break
	case 4
		TempValue0=Screen.CenterY
		TempValue0-=87
		if Object.YPos>TempValue0
			Object.YPos-=8
			if Object.YPos<TempValue0
				Object.YPos=TempValue0
			endif
		else
			Object.State++
			Object.Alpha=0
			Object[+4].Value1=0
		endif

	case 5
		if Object.Value0<256
			Object.Alpha=Object.Value0
			Object.Value0+=8
		else
			Object.Value0=0
			Object.Alpha=255
			Object.State++
		endif

			Object[+4].Value1--
			PlaySfx(23,0)
			if Object[+4].Value1<0
				Object[+4].Value1=2
			endif
		endif

			Object[+4].Value1++
			PlaySfx(23,0)
			if Object[+4].Value1>2
				Object[+4].Value1=0
			endif
		endif

		case 0
			if KeyPress.Left==1
				Object.Value3=10
				SaveRAM[33]-=10
				if SaveRAM[33]<0
					SaveRAM[33]=0
					Engine.BGMVolume=SaveRAM[33]
				else
					Engine.BGMVolume=SaveRAM[33]
					PlaySfx(23,0)
				endif
			endif
			if KeyPress.Right==1
				Object.Value4=10
				SaveRAM[33]+=10
				if SaveRAM[33]>100
					SaveRAM[33]=100
					Engine.BGMVolume=SaveRAM[33]
				else
					Engine.BGMVolume=SaveRAM[33]
					PlaySfx(23,0)
				endif
			endif
			break
		case 1
			if KeyPress.Left==1
				Object.Value5=10
				SaveRAM[34]-=10
				if SaveRAM[34]<0
					SaveRAM[34]=0
					Engine.SFXVolume=SaveRAM[34]
				else
					PlaySfx(23,0)

				endif
			endif
			if KeyPress.Right==1
				Object.Value6=10
				SaveRAM[34]+=10
				if SaveRAM[34]>100
					SaveRAM[34]=100
				else
					PlaySfx(23,0)
					Engine.SFXVolume=SaveRAM[34]
				endif
			endif
			break
		case 2
			if KeyPress.Left==1
				SaveRAM[35]--
				if SaveRAM[35]<0
					SaveRAM[35]=1
				endif
				PlaySfx(23,0)
			endif
			if KeyPress.Right==1
				SaveRAM[35]++
				if SaveRAM[35]>1
					SaveRAM[35]=0
				endif
				PlaySfx(23,0)
			endif
			break

		endswitch
		if KeyPress.ButtonB==1
			Object.State=9
			Object.Value0=248
			Options.OriginalControls=SaveRAM[35]
			SaveRAM[32]=1
			WriteSaveRAM()
			PlayStageSfx(1,0)
		endif
		break
	case 7
		if Object.Value0<30
			Object.Value0++
		else
			Object.Value0=0
			Object.State++
		endif
		break
	case 8
		if Object.Value0>0
			Object.Value0-=8
		else
			Object.State=1
			PlayMusic(0)
		endif
		TempValue0=Object.Value0
		if TempValue0>255
			TempValue0=255
		endif
		SetScreenFade(0,0,0,TempValue0)
		break
	case 9
		if Object.Value0>0
			Object.Alpha=Object.Value0
			Object.Value0-=8
		else
			Object.Alpha=0
			Object.State++
		endif
		break
	case 10
		TempValue0=Object.Value1
		TempValue0*=40
		TempValue0+=Screen.CenterY
		TempValue0-=87
		if Object.YPos<TempValue0
			Object.YPos+=8
			if Object.YPos>TempValue0
				Object.YPos=TempValue0
			endif
			Object.Alpha=0
		else
			Object.State=1
			Object.Alpha=255
		endif
		break
	case 11
		if Object.Value0>0
			Object.Alpha=Object.Value0
			Object.Value0-=8
		else
			Object.Type=TypeName[BlankObject]
			Object[-1].State=3
			Object[34].State=1
			Object[35].State=1
			Object[36].State=1
			Object[37].State=1
			Object[38].State=1

			Object[40].State=1
			Object[41].State=1
			Object[42].State=1
			Object[33].State=1
			Object[33].PropertyValue=5
		endif
		break
	case 12
		TempValue0=Screen.CenterY
		TempValue0-=87
		if Object.YPos>TempValue0
			Object.YPos-=8
			if Object.YPos<TempValue0
				Object.YPos=TempValue0
			endif
		else
			TempValue0=Object.EntityNo
			TempValue0+=10
			ResetObjectEntity(TempValue0,TypeName[AboutMenu],0,0,0)
			SetupMenu(MENU_1,0,3,1)
			GetVersionNumber(MENU_1,1)
			Object[+10].InkEffect=2
			Object.State++
			Object.Alpha=0
			Object[+4].Value1=0
		endif
		break
	case 13
		if Object.Value0<256
			Object.Alpha=Object.Value0
			Object.Value0+=8
		else
			Object.Value0=0
			Object.Alpha=255
			Object.State++
		endif
		break
	case 14
		if KeyPress.ButtonB==1
			Object.State=15
			Object.Value0=248
			PlayStageSfx(1,0)
		endif
		break
	case 15
		if Object.Value0>0
			Object.Alpha=Object.Value0
			Object.Value0-=8
		else
			Object.Alpha=0
			Object.State=10
			Object[+10].Type=TypeName[BlankObject]
		endif
		break

	endswitch
endsub
