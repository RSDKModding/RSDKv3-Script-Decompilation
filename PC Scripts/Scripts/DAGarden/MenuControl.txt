sub ObjectMain
	switch Object.State
	case 0
		Object.State=1
		Object.Value0=384
		SetScreenFade(0,0,0,255)
		PlayMusic(Options.Soundtrack)
		break
	case 1
		if Object.Value0>0
			Object.Value0-=8
		else
			Object.State=2
		endif
		SetScreenFade(0,0,0,Object.Value0)
		break
	case 2
		if KeyPress.Start==1
			StopMusic()
			Object.State=6
		endif
		if KeyPress.ButtonA==1
			Object.State=3
			Object[33].XPos=Screen.XSize
			Object[33].XPos+=96
			Object[33].XPos<<=16
			Object[33].State=1
			Object[33].DrawOrder=5
			Object[33].Priority=1
			Object[35].PropertyValue=4
			Object[36].PropertyValue=5
			Object[37].PropertyValue=6
		endif
		break
	case 3
		if Object[33].State==0
			if KeyPress.Left==1
				Object[33].State=2
			endif
			if KeyPress.Right==1
				Object[33].State=4
			endif
			if KeyPress.ButtonA==1
				Object.State=4
				StopMusic()
			endif
			if KeyPress.ButtonB==1
				Object.State=4
				StopMusic()
			endif
			if KeyPress.ButtonC==1
				Object.State=4
				StopMusic()
			endif
		endif
		if KeyPress.Start==1
			StopMusic()
			Object.State=6
		endif
		break
	case 4
		if Object.Value0<288
			Object.Value0+=8
		else
			Object[33].XPos=Screen.XSize
			Object[33].XPos+=96
			Object[33].XPos<<=16
			Object[33].State=0
			Object[33].Priority=0
			Object[35].PropertyValue=0
			Object[36].PropertyValue=2
			Object[37].PropertyValue=7
			CallFunction(MenuControl_Function0)
			Object.State++
		endif
		SetScreenFade(208,255,224,Object.Value0)
		break
	case 5
		if Object.Value0>0
			Object.Value0-=8
		else
			Object.State=2
		endif
		SetScreenFade(208,255,224,Object.Value0)
		break
	case 6
		if Object.Value0<384
			Object.Value0+=8
		else
			Stage.ListPos=1
			LoadStage()
		endif
		SetScreenFade(0,0,0,Object.Value0)
		break

	endswitch
endsub
